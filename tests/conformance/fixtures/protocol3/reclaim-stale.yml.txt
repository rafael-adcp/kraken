# Reaper for kraken. Copy to .github/workflows/reclaim-stale.yml in the
# coordination repo (see SKILL.md).
#
# A worker that dies mid-task leaves its issue labeled in-progress forever,
# blocking the queue. This workflow runs server-side (so it works even when every
# worker is dead): any in-progress issue whose worker has been silent for
# MAX_HOURS is moved to needs-decision for the human to triage — relaunch or
# investigate.
#
# Staleness is anchored to the worker's last liveness marker (the newest
# `claim`/`heartbeat` marker), NOT the issue's updatedAt. An operator
# comment on a dead worker's issue must shorten time-to-triage, not extend it —
# reading updatedAt would let any human comment keep a dead claim alive for
# another MAX_HOURS. An in-progress issue with no worker liveness marker at all is
# treated as infinitely stale and reclaimed.
#
# Workers must heartbeat: a short progress comment (the `heartbeat` marker)
# at least every ~2h on long tasks keeps the anchor fresh and the reaper away.
#
# Why not actions/stale: its threshold is whole DAYS (community-triage timescale);
# a dead worker's claim must be reclaimed within hours, not days.
name: Reclaim stale claims
on:
  schedule:
    - cron: "17 * * * *" # hourly; off-minute to dodge the top-of-hour crunch
  workflow_dispatch:

jobs:
  reclaim:
    runs-on: ubuntu-latest
    permissions:
      issues: write
    env:
      GH_TOKEN: ${{ github.token }}
      MAX_HOURS: 6
    steps:
      - name: Move silent in-progress issues to needs-decision
        run: |
          now=$(date -u +%s)
          # List ONLY the in-progress issue numbers (few by definition), then read
          # each thread's complete comment history separately. `gh issue list`'s
          # nested `comments` export silently caps at 100 and does not paginate —
          # the exact truncation class kraken.py's comment_bodies documents — so on
          # a thread past 100 comments the newest heartbeat: line falls outside the
          # export and a live, heartbeating worker looks silent, gets reclaimed, and
          # a second worker can claim a task still being executed. The paginated
          # REST comments endpoint (the same path comment_bodies uses) walks every
          # page, so the anchor is always computed against the whole history.
          #
          # Per issue, the anchor is the createdAt of the newest comment carrying a
          # worker liveness payload — a protocol/3 hidden marker
          # (<!-- kraken {"type":"claim"|"heartbeat", ...} -->) — or the literal
          # "none" when the worker never spoke. Only claim/heartbeat mark liveness:
          # delivered/released already drop in-progress, so they never appear on a
          # still-held claim. Operator comments are ignored — they must not reset
          # the clock.
          gh issue list -R "$GITHUB_REPOSITORY" --label in-progress --state open \
            --json number --jq '.[].number' |
          while read -r num; do
            [ -n "$num" ] || continue
            # --jq emits one createdAt per liveness comment across every page;
            # ISO-8601 UTC timestamps sort lexically = chronologically, so
            # `sort | tail -1` is the newest anchor. No match -> empty -> "none".
            anchor=$(gh api "repos/$GITHUB_REPOSITORY/issues/$num/comments" --paginate \
              --jq '.[] | select(.body | test("<!-- kraken \\{\"type\":\"(claim|heartbeat)\"")) | .createdAt' \
              | sort | tail -n1)
            [ -n "$anchor" ] || anchor="none"
            if [ "$anchor" = "none" ]; then
              age="" # no worker liveness marker at all: infinitely stale
            else
              age=$(( (now - $(date -u -d "$anchor" +%s)) / 3600 ))
              [ "$age" -lt "$MAX_HOURS" ] && continue
            fi
            reason=$([ -z "$age" ] && echo "no worker heartbeat on record" || echo "no worker heartbeat for ${age}h")
            gh issue edit "$num" -R "$GITHUB_REPOSITORY" \
              --remove-label in-progress --add-label needs-decision
            gh issue comment "$num" -R "$GITHUB_REPOSITORY" \
              --body "$(printf 'The worker has gone silent (%s) and likely died. To requeue, remove the needs-decision label; or investigate first.\n\n<!-- kraken {"type":"stale-claim","reason":"%s"} -->' "$reason" "$reason")"
          done
